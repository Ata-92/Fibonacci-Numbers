Python Version

rank = int(input("Please enter nth number: "))
number1 = 1
number2 = 2
fiboSequence = [number1, number2]
for i in range(rank):
    number1, number2 = number2, number1 + number2
    fiboSequence.append(number2)
print(fiboSequence[rank - 1])

let number2 = 2;
